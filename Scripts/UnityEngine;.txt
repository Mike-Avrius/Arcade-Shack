using UnityEngine;

public class TockenManager : MonoBehaviour
{
    public int tokens = 0;

    void Start()
    {
        LoadTockens();
    }

    public void UseTocken()
    {
        tokens = Mathf.Max(0, tokens - 1); // не даём уйти в минус
        SaveTockens();
    }

    public void LoadTockens()
    {
        if (!PlayerPrefs.HasKey("Tockens"))
        {
            tokens = 5; // первый запуск — 5 токенов
            SaveTockens();
        }
        else
        {
            tokens = PlayerPrefs.GetInt("Tockens");
        }
    }

    public void SaveTockens()
    {
        PlayerPrefs.SetInt("Tockens", tokens);
        PlayerPrefs.Save();
    }
    
}
