using UnityEngine;

public class BiverTimer : MonoBehaviour
{
    public Transform[] spawnPoints;
    public GameObject biverPrefab;

    private int lastSpawnIndex = -1; // сохраняем индекс последней позиции

    public void SpawnBivers()
    {
        int spawns;

        // Повторяем выбор, пока не найдём другой индекс
        do
        {
            spawns = Random.Range(0, spawnPoints.Length);
        } while (spawns == lastSpawnIndex && spawnPoints.Length > 1); 

        // Сохраняем текущий индекс
        lastSpawnIndex = spawns;

        Instantiate(biverPrefab, spawnPoints[spawns].position, Quaternion.identity);
    }

    
    // Start is called once before the first execution of Update after the MonoBehaviour is created
    void Start()
    {
        
    }

    // Update is called once per frame
    void Update()
    {
        
    }
}
